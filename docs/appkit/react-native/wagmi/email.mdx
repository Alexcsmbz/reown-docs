### Install packages

```
yarn add @reown/appkit-auth-wagmi-react-native react-native-webview
```

On iOS, use CocoaPods to add the native modules to your project:

```
npx pod-install
```

### Add the auth connector in `defaultWagmiConfig`

```ts
/* highlight-add-start */
import { AuthConnector } from '@reown/appkit-auth-wagmi-react-native'

const authConnector = AuthConnector({ projectId, metadata })
/* highlight-add-end */

const wagmiConfig = defaultWagmiConfig({
  chains,
  projectId,
  metadata,
  /* highlight-add-start */
  extraConnectors: [authConnector]
  /* highlight-add-end */
})
```
